# 模拟

Time: O(max(m, n)), 较长的链表的长度

Space: O(1), 2个node, 1个整数a

```python
class Solution:
    def addTwoNumbers(self, l1, l2):
        a = 0 # 记录当前位置的和, 并且记录进位, 每次循环结束只能是1或0
        head = ListNode(-1, None)
        cur = head
        while l1 != None or l2 != None or a != 0: # 2个数字或进位还没处理完
            if l1 != None:
                a += l1.val
                l1 = l1.next
            if l2 != None:
                a += l2.val
                l2 = l2.next
            cur.next = ListNode(a % 10, None)
            cur = cur.next
            a = 1 if a >= 10 else 0
        return head.next
```

注意题目要求输出: 两数之和的**逆序链表**.

很久之前写的Java模拟解法, 还需要反转链表, 相当于给自己增加难度.

```java
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode l = new ListNode() {};
        int a = 0;
        // add from lowest digit
        while ((l1 != null) || (l2 != null) || a != 0) {

            if (l1 != null) {
                a += l1.val;
                l1 = l1.next;
            }
            if (l2 != null) {
                a += l2.val;
                l2 = l2.next;
            }
            l = new ListNode(a % 10, l);
            a = (a > 9 ? 1 : 0);
        }
        // reverse
        ListNode reversed_l = new ListNode(l.val) {};
        ListNode current = l.next;
        while (current != null) {
            reversed_l = new ListNode(current.val, reversed_l);
            current = current.next;
        }
        return reversed_l.next;
    }
}
```
