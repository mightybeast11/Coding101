# 双指针：不用helper function



```python
class Solution:
    def validPalindrome(self, s: str) -> bool:
        low = 0
        high = len(s) - 1
        count = 1
        while low <= high:
            if low == high:
                return True
            elif low + 1 == high:
                if s[low] == s[high] or count == 1:
                    return True
            elif s[low] == s[high]:
                low += 1
                high -= 1
                continue
            elif s[low] != s[high] and count == 1:
                count = 0
                # 这一层的判断有问题。s[low+1]有可能恰好与s[high]相等，但是s[low]其实不是那个多余字符，具体错误见下面的test。这种情况我们还需要考虑s[low]与s[high-1]，而不应该直接返回False，但是使用纯迭代我没有想到实现方法，欢迎补充修改。
                if s[low+1] == s[high]:
                    low += 1
                    continue
                elif s[low] == s[high-1]:
                    high -= 1
                    continue
                else:
                    return False
            else:
                return False
        return False
```

# 双指针：用helper function

Time: O(n)

Space: O(1)

```python
class Solution:
    def validPalindrome(self, s: str) -> bool:
        
        def palindrome(s):
            low = 0
            high = len(s) - 1
            while low < high:
                if s[low] == s[high]:
                    low += 1
                    high -= 1
                else:
                    return False
            return True

        low = 0
        high = len(s) - 1
        while low < high:
            if s[low] == s[high]:
                low += 1
                high -= 1
            else:
                return palindrome(s[low+1:high+1]) or palindrome(s[low:high])
        return True
```
